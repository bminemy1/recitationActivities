\section{Interpreting Limits}
\label{section:interpreting-limits}

\begin{problem}
  \label{problem:conceptual-check-on-limit-definition}
  \mbox{}
  \outcome{Distinguish between limit values and function values.}
  \outcome{Identify the points where a limit does not exist and explain why.}
  \begin{itemize}
    \item[(a)]
      \textbf{True or False:}
      To find $\lim_{x \to 2} f(x)$, it's enough to know the values of $f(2.1)$, $f(2.01)$, $f(2.001)$, and so on.
      \begin{freeResponse}
        This statement is false!

        The given values will only help us make a guess at the right-sided limit of $f$ as $x$ approaches $2$, that is, $\lim_{x \to 2^+} f(x)$.
        To determine $\lim_{x \to 2} f(x)$, we also need to know $\lim_{x \to 2^-} f(x)$ which we cannot determine from the given values.
        (Why?)

        \textbf{Spoiler Alert:} For example, consider the function $f$ defined by
        \[
          f(x) =
        \begin{cases}
         -1 & \mbox{if $x < 2$,}\\
          0 & \mbox{if $x = 2$, and}\\
          1 & \mbox{if $2 < x$.}
        \end{cases}
        \]

       	Looking at the graph of this function
        \begin{center}
          \includegraphics[scale = 0.6]{Images/"Graph of piecewise defined function".png}
        \end{center}
        we see that $\lim_{x \to 2^+} f(x) = 1$ and $\lim_{x \to 2^-} f(x) = -1$.
        Thus, since $\lim_{x \to 2^+} f(x) \neq \lim_{x \to 2^-} f(x)$, $\lim_{x \to 2} f(x)$ does not exist.
      \end{freeResponse}

    \item[(b)]
      \textbf{True or False:}
       If we know the value of $f(2)$, then we know the value of $\lim_{x \to 2} f(x)$.
       \begin{freeResponse}
         This is also false!

         Using the example of part (a), $f(2) = 0$, while $\lim_{x \to 2} f(x)$ does not exist.
       \end{freeResponse}
  \end{itemize}
\end{problem}

\begin{problem}
  \label{problem:finding-limits-via-graphs}
  \outcome{Find limits from a graph (or state that the limit does not exist).}
  \outcome{Distinguish between limit values and function values.}
  Use the graphs and the given definitions of the following two functions to answer the questions below.
  \begin{image}
    \hspace*{-5em}
    \includegraphics{Images/"Graph of rational function".png}
    \hspace*{3em}
        \includegraphics{Images/"Graph of linear function".png}
  \end{image}
  \begin{itemize}
    \item[(a)]
      Find the domain of $f$ and the domain of $g$.
      \begin{freeResponse}
        The domain of $f$ is $(-\infty, 1) \cup (1, \infty)$ (all real numbers except $1$).
        The domain of $g$ is $(-\infty, \infty)$ (all real numbers).
      \end{freeResponse}

    \item[(b)]
      Is $f = g$?
      (Why or why not?)
      \begin{freeResponse}
        No, these two functions are not equal.
        Two functions are equal if and only if they have identical domains and their values agree on points in the domain.

        Since $f$ and $g$ have different domains, by part (a), they cannot be equal functions.
      \end{freeResponse}

    \item[(c)]
      Looking at the graphs, find $\lim_{x \to 1} f(x)$ and $\lim_{x \to 1} g(x)$.
      \begin{freeResponse}
        From the graphs we have that $\lim_{x \to 1} f(x) = 2$ and $\lim_{x \to 1} g(x) = 2$.
      \end{freeResponse}
  \end{itemize}
\end{problem}

\begin{problem}
  \outcome{Distinguish between limit values and function values.}
  \outcome{Define a one-sided limit.}
  \outcome{Identify the points where a limit does not exist and explain why.}
  \textbf{True or False}: Give an explanation or counterexample.
  Assume $a$ and $L$ are finite numbers.
  \begin{itemize}
    \item[(a)]
      If $ \lim_{x \to a} f(x) = L$, then $f(a) = L$.
      \begin{freeResponse}
        This statement is false.
        See the example in problem 1(a).
      \end{freeResponse}

    \item[(b)]
      If $  \lim_{x \to a^-} f(x) = L$, then $  \lim_{x \to a^+} f(x) = L $.
      \begin{freeResponse}
        This statement is also false.
        See the example in problem 1(a).
      \end{freeResponse}

    \item[(c)]
      If $ \lim_{x \to a} f(x) = L $ and $  \lim_{x \to a} g(x) = L $, then $f(a) = g(a)$.
      \begin{freeResponse}
        This statement is false.
        See problem 2.
      \end{freeResponse}

    \item[(d)]
       $ \lim_{x \to a} \frac{f(x)}{g(x)} $ does not exist if $g(a) = 0$.
       \begin{freeResponse}
         This statement is false.
         See problem 2.
       \end{freeResponse}
  \end{itemize}
\end{problem}
